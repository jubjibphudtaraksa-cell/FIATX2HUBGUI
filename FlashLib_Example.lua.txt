Stroke.Color = Color3.fromRGB(255, 255, 255)

local ClickButton = Instance.new("TextButton")

ClickButton.Parent = Button

-- ═══════════════════════════════
--          ลากปุ่มได้
-- ═══════════════════════════════
local dragging = false
local dragStart
local startPos
local wasDragging = false
local dragThreshold = 6

ClickButton.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or
       input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        wasDragging = false
        dragStart = input.Position
        startPos = Button.Position
    end
end)

ClickButton.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or
       input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if dragging and (
        input.UserInputType == Enum.UserInputType.Touch or
        input.UserInputType == Enum.UserInputType.MouseMovement
    ) then
        local delta = input.Position - dragStart
        if delta.Magnitude > dragThreshold then
            wasDragging = true
        end
        Button.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

-- ═══════════════════════════════
--           สร้าง Window
-- ═══════════════════════════════
local Window = FlashLib:Window({
    Title = "My Script",
    Subtitle = "by Me",
    Size = UDim2.fromOffset(868, 650),
})

local TweenService = game:GetService("TweenService")
local LOGO_OPEN   = "rbxassetid://10734950309"
local LOGO_CLOSED = "rbxassetid://18821914323"
local isOpen = true

local function SetGuiState(state)
    isOpen = state
    Window.Settings = Window.Settings or {}
    
    Logo.Image = state and LOGO_OPEN or LOGO_CLOSED
    
    TweenService:Create(Stroke, TweenInfo.new(0.2, Enum.EasingStyle.Sine), {
        Color = state
            and Color3.fromRGB(100, 200, 255)
            or Color3.fromRGB(255, 255, 255),
        Transparency = state and 0.3 or 0.7
    }):Play()
    
    TweenService:Create(Logo, TweenInfo.new(0.2, Enum.EasingStyle.Sine), {
        ImageTransparency = state and 0 or 0.3
    }):Play()
    
    TweenService:Create(Button, TweenInfo.new(0.2, Enum.EasingStyle.Sine), {
        BackgroundColor3 = state
            and Color3.fromRGB(20, 30, 40)
            or Color3.fromRGB(15, 15, 15)
    }):Play()
end

-- กดปุ่มสลับเปิดปิด (กันลากแล้วเปิดโดยไม่ตั้งใจ)
local lastClick = 0
ClickButton.MouseButton1Click:Connect(function()
    if wasDragging then
        wasDragging = false
        return
    end
    local now = tick()
    if now - lastClick < 0.2 then return end
    lastClick = now
    SetGuiState(not isOpen)
end)

-- Touch สำหรับมือถือ
ClickButton.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch then
        if wasDragging then
            wasDragging = false
            return
        end
        local now = tick()
        if now - lastClick < 0.2 then return end
        lastClick = now
        SetGuiState(not isOpen)
    end
end)
