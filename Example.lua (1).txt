--// UILib Example Usage for Roblox
local UILib = require(script.UILib)

--// Create Main Window
local Window = UILib:Window({
	Title = "MacLib Style UI",
	Subtitle = "This is a subtitle",
	Size = UDim2.fromOffset(868, 650),
})

--// Create Tabs
local DemoTab = Window:AddTab("Demo", "rbxassetid://18821914323")
local SettingsTab = Window:AddTab("Settings", "rbxassetid://10734950309")
local CreatorsTab = Window:AddTab("Creators", "rbxassetid://18821914323")

--// === DEMO TAB ===
print("Building Demo Tab...")

--// Demo Tab - Button Section
local demoSection = DemoTab:AddSection()
demoSection:AddLabel("Button Example")
demoSection.Frame:FindFirstChildOfClass("UIListLayout").Parent.Size = UDim2.new(1, 0, 0, 40)

local buttonDemo = DemoTab:AddButton("Click Me!", function()
	Window:Notify("Button Clicked! 🎉", 3)
end)

--// Demo Tab - Toggle Section
local toggleDemo = DemoTab:AddToggle("Enable Feature", function(state)
	Window:Notify((state and "✓ Enabled" or "✗ Disabled") .. " Feature", 2)
end)

--// Demo Tab - Slider Section
local sliderDemo = DemoTab:AddSlider("Volume", 0, 100, 50, function(value)
	print("Volume: " .. value)
end)

--// Demo Tab - Dropdown Section
local dropdownOptions = {"Option 1", "Option 2", "Option 3", "Option 4", "Option 5"}
local dropdownDemo = DemoTab:AddDropdown("Choose", dropdownOptions, function(selected)
	Window:Notify("Selected: " .. selected, 2)
end)

--// Demo Tab - TextBox Section
local textboxDemo = DemoTab:AddTextBox("Username", "Enter your name...", function(text)
	if text ~= "" then
		Window:Notify("Hello, " .. text .. "!", 2)
	end
end)

--// === SETTINGS TAB ===
print("Building Settings Tab...")

local settingsSection = SettingsTab:AddSection()
settingsSection:AddLabel("Settings")

local uiBlurToggle = SettingsTab:AddToggle("UI Blur", function(state)
	Window:Notify((state and "Enabled" or "Disabled") .. " UI Blur", 2)
end)

local notificationToggle = SettingsTab:AddToggle("Notifications", function(state)
	Window:Notify((state and "Enabled" or "Disabled") .. " Notifications", 2)
end)

local themeDropdown = SettingsTab:AddDropdown("Theme", {"Dark", "Light", "Auto"}, function(selected)
	Window:Notify("Theme set to: " .. selected, 2)
end)

--// === CREATORS TAB ===
print("Building Creators Tab...")

local creatorSection = CreatorsTab:AddSection()
creatorSection:AddLabel("Credits")

CreatorsTab:AddButton("Creator 1", function()
	Window:Notify("Visit Creator 1", 2)
end)

CreatorsTab:AddButton("Creator 2", function()
	Window:Notify("Visit Creator 2", 2)
end)

CreatorsTab:AddButton("Creator 3", function()
	Window:Notify("Visit Creator 3", 2)
end)

--// Notifications
Window:Notify("UILib Loaded Successfully! 🚀", 5)

print("UI Ready!")
